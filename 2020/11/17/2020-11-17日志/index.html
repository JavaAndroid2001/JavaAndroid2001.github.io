/*
张三一个月生活费800
如果他的钱花到少于200就去搬砖
搬砖一天+100
记录张三这个月的支出和收入

   1.程序可以一直运行，除非收到退出指令
   2.当余额少于200的时候，输出3句去搬砖的提示

   解决方法：
        1.成员变量
            1.1 张三初始化的钱数量，张三的天数
        2.成员方法
            2.1 张三每天花钱
            2.2 张三每天搬砖能 +100
            2.3 程序是一直运行的，除非接收到退出指令
            2.4 余额少于200时，输出3句去搬砖的提示

 */


  

已完善：

public static void main(String[] args) {
  int day = 0;
  int money = 800;
  int exp = 0;      //总支出
  int income = 0;     //总收入
  String work = null;
  Scanner input = new Scanner(System.in);
  Random r = new Random();
  do {
      System.out.println("请输入：消费、打工、退出");
      work = input.next();
      if (work.equals("打工")) {
          money += 100;           //打工+100
          int im = money;         //这次打工的收入
          income = im;            //把这次的钱传给总收入
          int f1 = money;
          System.out.println("您打工挣得" + f1);
          System.out.println("您总资产现为：" + money);
      } else if (work.equals("消费")) {
          if (money >= 200) {
              int iq = 0;

              int consumption = r.nextInt(801);
              money -= consumption;        //随机消费
              iq = consumption;           //这次消费的钱传给一个储存变量
              exp = iq;                   //把这次的钱传给总支出
              System.out.println("这次消费了：" + consumption);
              System.out.println("您总资产现为：" + money);

          }else {
              System.out.println("您的余额已不足200元，总资产为：" + money);
              System.out.println("打工去吧");
              System.out.println("打工去吧");
              System.out.println("打工去吧");

          }


      } else if (work.equals("退出")) {
          System.out.println("程序已经退出！");
          System.exit(0);     //程序退出
          input.close();          //结束输入
      }

     
      if (day < 30){
          day += 1;
          System.out.println("今天是第："+day+"天");
      }else {
          day = 0;
          System.out.println("月底了，这个月总支出："+ exp);
          exp = 0;
          System.out.println("总收入为："+ income);
          income = 0;
      }
  }while (true);

  }
